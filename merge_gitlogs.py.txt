import pandas as pd

# Load Git logs from CSV files
df_sophie = pd.read_csv("gitlog_sophie.csv", names=["Commit Hash", "Author", "Date", "Message"])
df_serena = pd.read_csv("gitlog_serena.csv", names=["Commit Hash", "Author", "Date", "Message"])
df_komal = pd.read_csv("gitlog_komal.csv", names=["Commit Hash", "Author", "Date", "Message"])

# Add a column to indicate the branch
df_sophie["Branch"] = "Sophie"
df_serena["Branch"] = "Serena"
df_komal["Branch"] = "Komal"

# Merge all three DataFrames into one
df_combined = pd.concat([df_sophie, df_serena, df_komal])

# Save to an Excel file
output_filename = "gitlog_combined.xlsx"
df_combined.to_excel(output_filename, index=False, engine="openpyxl")

print("Git log exported to gitlog_combined.xlsx successfully!")
